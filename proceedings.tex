\newcommandtwoopt{\addpaper}[5][DEF][DEF]{%

    \fancypagestyle{lniotherpages}{%
        \fancyhead{}
        %title on the right
        \fancyhead[RO]{\small \ifthenelse{\equal{#2}{DEF}}{#5}{#2}\hspace{5pt}\thepage\hspace{0.05cm}}
        %authors on the left
        \fancyhead[LE]{\hspace{0.05cm}\small\thepage\hspace{5pt}\ifthenelse{\equal{#1}{DEF}}{#4}{#1}}
        \fancyfoot{} % Löscht alle Fußzeileneinstellungen
        \renewcommand{\headrulewidth}{0.4pt} %Linie unter Kopfzeile
    }

    %collect page number
    \write\pagesfile{#3 \thepage}
    \setcounter{startpage}{\value{page}}

    \pdfximage{papers/#3/paper.pdf}

    \thispagestyle{lnifirstpage}
    \AddToShipoutPictureBG*{\AtPageUpperLeft{%
        %When using 23.649cm, there might be the toprule shown - try different zoom levels and Acrobat. With SumatraPDF, the issue is not visible all times
        \put(\LenToUnit{0pt},\LenToUnit{-29.7cm}){\includegraphics[page=1, viewport=0 0 21cm 23.628cm, clip]{papers/#3/paper.pdf}}}
    }
	\phantomsection
    \suppress\cite{#3}\endsuppress
    \addcontentsline{toc}{section}{\lnitoc{#4}{#5}}
    \newpage
    \pagestyle{lniotherpages}

    \ifthenelse{\the\pdflastximagepages>1}{
        \setcounter{curpage}{1}
        \whiledo{\value{curpage}<\the\pdflastximagepages}{
            \stepcounter{curpage}
            \AddToShipoutPictureBG*{\AtPageUpperLeft{%
              \put(\LenToUnit{0pt},\LenToUnit{-29.7cm}){\includegraphics[page=\the\value{curpage}, viewport=0 0 21cm 23.628cm, clip]{papers/#3/paper.pdf}}}
            }
            \null
            \newpage
        }
    }

    {
        \renewcommand{\and}{ and }
        \addtostream{authorsbib}{@inproceedings{#3,^^J  author = {#4},^^J  title = {#5},^^J  booktitle = {\bibbooktitle},^^J  pages = {\thestartpage--\the\numexpr\value{page}-1\relax},^^J  eventdate = {\eventdate},^^J  publisher = {Gesellschaft f\"{u}r Informatik e.V.},^^J  year = {\proceedingsyear},^^J}^^J}
    }

	{
        \renewcommand{\and}{;}
        \addtostream{proceedingscsv}{#4;#5;\thestartpage--\the\numexpr\value{page}-1\relax}
    }

    \cleardoublepage

} %END of \newcommand{addpaper}
